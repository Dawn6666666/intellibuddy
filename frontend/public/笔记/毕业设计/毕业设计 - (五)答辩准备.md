### 5.1 答辩PPT制作

```python
class DefensePPT:
    def __init__(self, project_title):
        self.project_title = project_title
        self.slides = []
    
    def create_defense_slides(self):
        """创建答辩PPT结构"""
        
        slides = [
            {
                'title': '封面',
                'content': [
                    '论文题目',
                    '学生姓名、学号',
                    '指导教师',
                    '答辩日期'
                ],
                'time': '30秒'
            },
            {
                'title': '目录',
                'content': [
                    '研究背景',
                    '系统设计',
                    '实现过程',
                    '测试结果',
                    '总结展望'
                ],
                'time': '30秒'
            },
            {
                'title': '研究背景',
                'content': [
                    '问题提出',
                    '研究意义',
                    '国内外现状',
                    '本文工作'
                ],
                'time': '2分钟'
            },
            {
                'title': '系统设计',
                'content': [
                    '需求分析',
                    '总体架构',
                    '功能模块',
                    '技术选型'
                ],
                'time': '3分钟'
            },
            {
                'title': '关键技术',
                'content': [
                    '核心算法',
                    '技术难点',
                    '解决方案',
                    '创新点'
                ],
                'time': '3分钟'
            },
            {
                'title': '系统实现',
                'content': [
                    '开发环境',
                    '核心代码',
                    '界面展示',
                    '功能演示'
                ],
                'time': '4分钟'
            },
            {
                'title': '测试与分析',
                'content': [
                    '测试方案',
                    '性能测试',
                    '对比分析',
                    '结果评价'
                ],
                'time': '2分钟'
            },
            {
                'title': '总结与展望',
                'content': [
                    '完成工作',
                    '创新贡献',
                    '存在不足',
                    '改进方向'
                ],
                'time': '2分钟'
            },
            {
                'title': '谢谢聆听',
                'content': [
                    '请各位老师批评指正'
                ],
                'time': '30秒'
            }
        ]
        
        self.slides = slides
    
    def print_slide_outline(self):
        """打印PPT大纲"""
        print(f"🎯 《{self.project_title}》答辩PPT大纲")
        print("=" * 50)
        
        total_time = 0
        
        for i, slide in enumerate(self.slides, 1):
            print(f"\n第{i}页: {slide['title']} ({slide['time']})")
            for item in slide['content']:
                print(f"  • {item}")
            
            # 计算时间
            time_str = slide['time']
            if '分钟' in time_str:
                minutes = int(time_str.replace('分钟', ''))
                total_time += minutes
            elif '秒' in time_str:
                seconds = int(time_str.replace('秒', ''))
                total_time += seconds / 60
        
        print(f"\n⏱️  总时长: {total_time:.1f}分钟")
        print("\n💡 PPT制作建议:")
        print("  • 每页内容不超过7行")
        print("  • 字体大小不小于24号")
        print("  • 配色简洁，对比度高")
        print("  • 动画效果适度，不喧宾夺主")
        print("  • 准备演示视频，防止现场故障")

# PPT大纲生成
ppt = DefensePPT("基于深度学习的智能推荐系统设计与实现")
ppt.create_defense_slides()
ppt.print_slide_outline()
```

                'A: 说明测试方法和验证标准',

                'A: 从技术价值和商业价值两方面回答',

                'A: 提供实验结果和用户反馈'
            ],

                'Q: 系统还有哪些不足？',
                'A: 客观分析不足，提出改进方向',
                
                'Q: 如何进一步完善系统？',
                'A: 提出具体的改进计划和实施方案',

                'A: 分析市场前景和商业模式',
                
                'Q: 对这个领域的发展有什么看法？',
                'A: 结合技术趋势和行业发展回答'
            ]
        }
    
    def print_qa_guide(self):
        """打印问答指南"""
        print("❓ 答辩常见问题与回答策略")
        print("=" * 50)
        
        for category, qa_list in self.questions.items():
            print(f"\n📋 {category}:")
            
            for i in range(0, len(qa_list), 2):
                question = qa_list[i]
                answer = qa_list[i + 1] if i + 1 < len(qa_list) else ""
                
                print(f"\n  {question}")
                print(f"  {answer}")
    
    def defense_tips(self):
        """答辩技巧"""
        tips = [
            '🎯 充分准备，熟悉自己的系统',

            '💡 不知道的问题要诚实承认',
            '⏰ 控制时间，重点突出',
            '🤝 尊重老师，态度谦逊',
            '📊 准备演示，展示系统功能',
            '🔧 准备备用方案，防止技术故障'
        ]
        
        print("\n💡 答辩技巧:")
        for tip in tips:
            print(f"  {tip}")
