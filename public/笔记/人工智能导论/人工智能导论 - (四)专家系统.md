### 4.1 规则引擎

```python
class Rule:
    def __init__(self, conditions, conclusion):
        self.conditions = conditions
        self.conclusion = conclusion
    
    def matches(self, facts):
        """检查规则是否匹配"""
        return all(fact in facts for fact in self.conditions)

class ExpertSystem:
    def __init__(self):
        self.rules = []
        self.facts = set()
    
    def add_rule(self, conditions, conclusion):
        """添加规则"""
        self.rules.append(Rule(conditions, conclusion))
    
    def add_fact(self, fact):
        """添加事实"""
        self.facts.add(fact)
    
    def infer(self):
        """推理"""
        changed = True
        while changed:
            changed = False
            for rule in self.rules:
                if rule.matches(self.facts) and rule.conclusion not in self.facts:
                    self.facts.add(rule.conclusion)
                    print(f"推导出: {rule.conclusion}")
                    changed = True
        
        return self.facts

# 示例：动物识别系统
es = ExpertSystem()

# 添加规则
es.add_rule(['有毛发', '哺乳动物'], '是哺乳动物')
es.add_rule(['有羽毛'], '是鸟类')
es.add_rule(['是哺乳动物', '吃肉'], '是食肉动物')
es.add_rule(['是食肉动物', '有斑点'], '是豹子')

# 添加事实
es.add_fact('有毛发')
es.add_fact('哺乳动物')
es.add_fact('吃肉')
es.add_fact('有斑点')

# 推理
result = es.infer()
print(f"最终结论: {result}")
```
