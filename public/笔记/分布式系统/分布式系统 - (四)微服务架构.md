### 4.1 服务发现

```python
import time
import threading

class ServiceRegistry:
    def __init__(self):
        self.services = {}  # service_name -> [(host, port, last_heartbeat), ...]
        self.lock = threading.Lock()
        self.heartbeat_timeout = 30  # 30秒超时
    
    def register(self, service_name, host, port):
        """注册服务"""
        with self.lock:
            if service_name not in self.services:
                self.services[service_name] = []
            
            # 更新或添加服务实例
            instance = (host, port, time.time())
            
            # 检查是否已存在
            for i, (h, p, _) in enumerate(self.services[service_name]):
                if h == host and p == port:
                    self.services[service_name][i] = instance
                    return
            
            self.services[service_name].append(instance)
            print(f"✅ 注册服务: {service_name} at {host}:{port}")
    
    def deregister(self, service_name, host, port):
        """注销服务"""
        with self.lock:
            if service_name in self.services:
                self.services[service_name] = [
                    (h, p, t) for h, p, t in self.services[service_name]
                    if not (h == host and p == port)
                ]
                print(f"❌ 注销服务: {service_name} at {host}:{port}")
    
    def discover(self, service_name):
        """发现服务"""
        with self.lock:
            if service_name not in self.services:
                return []
            
            # 过滤掉超时的服务
            current_time = time.time()
            active_services = [
                (host, port) for host, port, last_heartbeat 
                in self.services[service_name]
                if current_time - last_heartbeat < self.heartbeat_timeout
            ]
            
            return active_services
    
    def heartbeat(self, service_name, host, port):
        """心跳检测"""
        self.register(service_name, host, port)  # 更新时间戳

class LoadBalancer:
    def __init__(self, registry):
        self.registry = registry
        self.round_robin_index = {}
    
    def get_instance(self, service_name, strategy='round_robin'):
        """获取服务实例"""
        instances = self.registry.discover(service_name)
        
        if not instances:
            return None
        
        if strategy == 'round_robin':
            if service_name not in self.round_robin_index:
                self.round_robin_index[service_name] = 0
            
            index = self.round_robin_index[service_name]
            instance = instances[index % len(instances)]
            self.round_robin_index[service_name] += 1
            
            return instance
        
        elif strategy == 'random':
            import random
            return random.choice(instances)

# 测试服务发现
registry = ServiceRegistry()
lb = LoadBalancer(registry)

# 注册服务
registry.register('user-service', '192.168.1.10', 8080)
registry.register('user-service', '192.168.1.11', 8080)
registry.register('order-service', '192.168.1.20', 8081)

# 服务发现
print("🔍 发现user-service:")
for _ in range(4):
    instance = lb.get_instance('user-service')
    print(f"  -> {instance}")

print("\n🔍 发现order-service:")
instance = lb.get_instance('order-service')
print(f"  -> {instance}")
```

### 4.2 熔断器模式

```python
import time
import random
from enum import Enum

class CircuitState(Enum):
    CLOSED = "CLOSED"      # 正常状态
    OPEN = "OPEN"          # 熔断状态
    HALF_OPEN = "HALF_OPEN"  # 半开状态

class CircuitBreaker:
    def __init__(self, failure_threshold=5, timeout=60, success_threshold=3):
        self.failure_threshold = failure_threshold  # 失败阈值
        self.timeout = timeout  # 熔断超时时间
        self.success_threshold = success_threshold  # 半开状态成功阈值
        
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = None
    
    def call(self, func, *args, **kwargs):
        """调用被保护的函数"""
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time > self.timeout:
                self.state = CircuitState.HALF_OPEN
                self.success_count = 0
                print("🔄 熔断器进入半开状态")
            else:
                raise Exception("熔断器开启，拒绝调用")
        
        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        
        except Exception as e:
            self._on_failure()
            raise e
    
    def _on_success(self):
        """成功回调"""
        if self.state == CircuitState.HALF_OPEN:
            self.success_count += 1
            if self.success_count >= self.success_threshold:
                self.state = CircuitState.CLOSED
                self.failure_count = 0
                print("✅ 熔断器关闭，恢复正常")
        else:
            self.failure_count = 0
    
    def _on_failure(self):
        """失败回调"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.OPEN
            print("💥 半开状态失败，重新熔断")
        elif self.failure_count >= self.failure_threshold:
            self.state = CircuitState.OPEN
            print(f"💥 失败次数达到阈值({self.failure_threshold})，熔断器开启")

# 模拟不稳定的服务
def unreliable_service():
    """不稳定的服务（70%失败率）"""
    if random.random() < 0.7:
        raise Exception("服务调用失败")
    return "服务调用成功"

# 测试熔断器
cb = CircuitBreaker(failure_threshold=3, timeout=5)

print("🧪 测试熔断器:")
for i in range(10):
    try:
        result = cb.call(unreliable_service)
        print(f"第{i+1}次调用: {result}")
    except Exception as e:
        print(f"第{i+1}次调用: {e}")
    
    time.sleep(0.5)

print(f"\n当前熔断器状态: {cb.state.value}")
```
